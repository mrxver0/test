local OrionLib = loadstring(game:HttpGet(('https://raw.githubusercontent.com/shlexware/Orion/main/source')))()
local Window = OrionLib:MakeWindow({Name = "SussyScrips PetGo Scripts 🐾🎮", HidePremium = false, SaveConfig = true, ConfigFolder = "Orion"})

local Pets = require(game:GetService("ReplicatedStorage").Library.Items.PetItem).Directory
local fromPet = "Cat"
local toPet = "None"
local selectedHugePet = "None"

function ChangeVisual()
    if toPet ~= "None" then
        for i, v in pairs(Pets[toPet] or {}) do
            Pets[fromPet][i] = v
        end
    end

    if selectedHugePet ~= "None" then
        for i, v in pairs(Pets[selectedHugePet] or {}) do
            Pets[fromPet][i] = v
        end
    end
    OrionLib:MakeNotification({
        Name = "Pet Spawner 🐾",
        Content = "Successfully Spawn Pet! ✨",
        Image = "rbxassetid://4483345998",
        Time = 5
    })

    AutoEquipCat()
end

function AutoEquipCat()
    local equipLimit = 5
    local equipCount = 0

    while equipCount < equipLimit do
        local args = {
            [1] = "93fbe670e652405181d49799250ce392"
        }
        game:GetService("ReplicatedStorage"):WaitForChild("Network"):WaitForChild("Pets_Equip"):FireServer(unpack(args))
        equipCount += 1
        task.wait(0.9)
    end
    OrionLib:MakeNotification({
        Name = "Auto Equip 🐾",
        Content = "Successfully equipped the Cat pet 5 times! 🐱",
        Image = "rbxassetid://4483345998",
        Time = 5
    })
end

local buying = false
local fishingMerchantNumber = 1
local standardMerchantNumber = 1
local potionVendingMachine = false
local buyLoop

local function startAutoBuyMerchants()
    buying = true
    buyLoop = task.spawn(function()
        while buying do
            fishingMerchantNumber = fishingMerchantNumber + 1
            if fishingMerchantNumber > 15 then
                fishingMerchantNumber = 1
            end

            local fishingArgs = {
                [1] = "FishingMerchant",
                [2] = fishingMerchantNumber
            }

            pcall(function()
                game:GetService("ReplicatedStorage"):WaitForChild("Network"):WaitForChild("CustomMerchants_Purchase"):InvokeServer(unpack(fishingArgs))
            end)

            standardMerchantNumber = standardMerchantNumber + 1
            if standardMerchantNumber > 10 then
                standardMerchantNumber = 1
            end

            local standardArgs = {
                [1] = "StandardMerchant",
                [2] = standardMerchantNumber
            }

            pcall(function()
                game:GetService("ReplicatedStorage"):WaitForChild("Network"):WaitForChild("CustomMerchants_Purchase"):InvokeServer(unpack(standardArgs))
            end)

            if potionVendingMachine then
                local potionArgs = {
                    [1] = "PotionVendingMachine"
                }

                pcall(function()
                    game:GetService("ReplicatedStorage"):WaitForChild("Network"):WaitForChild("VendingMachines_Purchase"):InvokeServer(unpack(potionArgs))
                end)
            end

            task.wait(0.2)
        end
    end)
    OrionLib:MakeNotification({
        Name = "Auto Merchant Buy 🛒",
        Content = "Successfully started Auto Merchant Buy! 💸",
        Image = "rbxassetid://4483345998",
        Time = 5
    })
end

local function stopAutoBuyMerchants()
    buying = false
    if buyLoop then
        task.cancel(buyLoop)
        buyLoop = nil
    end
    OrionLib:MakeNotification({
        Name = "Auto Merchant Buy 🛒",
        Content = "Successfully stopped Auto Merchant Buy! ❌",
        Image = "rbxassetid://4483345998",
        Time = 5
    })
end
