            end

                [1] = "FishingMerchant",
                [2] = fishingMerchantNumber

            pcall(function()
                game:GetService("ReplicatedStorage"):WaitForChild("Network"):WaitForChild("CustomMerchants_Purchase"):InvokeServer(unpack(fishingArgs))

                game:GetService("ReplicatedStorage"):WaitForChild("Network"):WaitForChild("CustomMerchants_Purchase"):InvokeServer(unpack(standardArgs))
